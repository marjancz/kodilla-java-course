SELECT L.NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUESLIST L, ISSUES I
WHERE L.ID = I.ISSUESLISTS_ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.LASTNAME, U.FIRSTNAME
FROM ISSUES I, USERS U
WHERE I.USER_ID_ASSIGNEDTO = U.ID

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_COUNT
FROM ISSUES I, USERS U
WHERE I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME, U.FIRSTNAME;
